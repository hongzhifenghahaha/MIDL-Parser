   ```java
   package org.example;
   
   import org.antlr.v4.runtime.ANTLRInputStream;
   import org.antlr.v4.runtime.CharStreams;
   import org.antlr.v4.runtime.CommonTokenStream;
   import org.antlr.v4.runtime.Parser;
   import org.antlr.v4.runtime.tree.ParseTree;
   import org.example.parser.MIDLLexer;
   import org.example.parser.MIDLParser;
   import org.junit.jupiter.api.Test;
   
   import java.io.*;
   import java.lang.reflect.InvocationTargetException;
   import java.net.URISyntaxException;
   import java.util.stream.Stream;
   
   public class MIDLTest {
       MIDLLexer lexer;
       CommonTokenStream tokenStream;
       Parser parser;
   
       public void testLine(String file, String ruleName) throws IOException, NoSuchMethodException, InvocationTargetException, IllegalAccessException, URISyntaxException {
           var path = this.getClass().getResource(file).toURI().getPath();
           FileInputStream input = new FileInputStream(path);
           BufferedReader bufferedReader = new BufferedReader(new InputStreamReader(input));
           String str;
           while((str=bufferedReader.readLine()) != null) {
               lexer = new MIDLLexer(CharStreams.fromString(str));
               tokenStream = new CommonTokenStream(lexer);
               parser = new MIDLParser(tokenStream);
               var parserClass = MIDLParser.class;
               var ruleMethod = parserClass.getMethod(ruleName);
               ruleMethod.invoke(parser);
           }
       }
   
       @Test
       public void testLiteral() throws IOException, InvocationTargetException, NoSuchMethodException, IllegalAccessException, URISyntaxException {
           testLine("/test_literal.txt", "literal");
       }
   }
   
   ```
